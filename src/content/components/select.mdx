## Examples

### Filled

#### Default

<ComponentContainer>
  <FilledSelectDefault slot='preview' client:visible/>

```tsx
import { FilledSelect as Select } from '@/components/ui/select'

export const FilledSelectDefault = () => {
  return (
    <Select>
      <Select.Trigger>
        <Select.Value placeholder="Select a city" />
      </Select.Trigger>
      <Select.Content>
        <Select.Group>
          <Select.Subheader>Andalucía</Select.Subheader>
          <Select.Item value="almeria">Almería</Select.Item>
          <Select.Item value="cadiz">Cádiz</Select.Item>
          <Select.Item value="cordoba">Córdoba</Select.Item>
          <Select.Item value="granada">Granada</Select.Item>

          <Select.Item value="jaen">Jaén</Select.Item>
          <Select.Item value="malaga">Málaga</Select.Item>
          <Select.Item value="sevilla">Sevilla</Select.Item>
        </Select.Group>
      </Select.Content>
    </Select>
  )
}
```

</ComponentContainer>
